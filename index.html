Welcome to GitHub Pages.

Log2server is a simple nodejs app to save log4j or log4net logs on the mongodb without need any change on your code. You just net change your appender configuration just like follows.

<?xml version="1.0" encoding="utf-8" ?>
<log4net>
  <appender name="UdpAppender" type="log4net.Appender.UdpAppender">
    <remoteAddress value="127.0.0.1" />
    <remotePort value="43278" />
    <layout type="log4net.Layout.XmlLayoutSchemaLog4j" />
  </appender>
  <root>
    <level value="ALL" />
    <appender-ref ref="UdpAppender" />
  </root>
</log4net>
